#### 1. `10^9` use long long int

Basic template for long long
```cpp
#include <iostream>
#include <vector>
using namespace std;

#define ll long long
#define int ll

int32_t main() {
    int t; cin >> t;
    while (t--) {
            int n; cin>>n;
            vector<int> a(n);
            for(int i=0;i<n;i++) cin>>a[i];
            int count=0;
            int temp=0;
            for(int i=0;i<n;i++){
                int x; cin>>x;
                if(x<=(a[i]-temp)) count++;
                temp = a[i];
            }
            cout<<count<<endl;
    }
    return 0;
}
```

#### 4. COOL NAME CODECHEF
```cpp
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
     string s; cin>>s;
     sort(s.begin(), s.end());
     int sum=0;
     for(int i=0; i<s.size(); i++){
       int pos = int(s[i])-96;
       int ans = pos*(i+1);
       sum += ans;
      }
      cout<<sum<<endl;
    }
    return 0;
}
```


#### 5. Non-Negative Product
```cpp
#include<bits/stdc++.h>
#define ll long long
#define int ll
using namespace std;

int32_t main()
{
    int t,n,x;
    cin >> t ;
    while(t--) {
        cin >> n;
        bool flag = false;
        int count = 0;
        for(int i=0; i<n; i++) {
            cin >> x ;
            if(x<0) {
                count++;
            }
            if(x==0) {
                flag = true;
            }
        }
        if(flag) {
            cout << 0 << endl;
        }
        else if (count%2==0) {
            cout << 0 << endl;
        }
        else {
            cout << 1 << endl;
        }
    }
    return 0;
}

```


#### 6. Two Different Palindromes
```cpp
#include<iostream>
using namespace std;

int main()
{
    int t,x,y;
    cin >> t;
    while(t--) {
        cin >> x >> y;
        if( (x%2!=0) && (y%2!=0) ) {
            cout << "NO" << endl;
        }
        else if ( (x==1) || (y==1) ) {
            cout << "NO" << endl;
        }
        else {
            cout << "YES" << endl;
        }
    }
}
```

#### 11. string : I am smith => smith am I
```cpp
#include<bits/stdc++.h>
using namespace std;

int32_t main()
{
    // input
    string str;
    getline(cin,str);
    
    int t = str.size();
    map<char,int> mp;
    
    // processing
    for(int i=0; i<t; i++) {
        if(str[i]!=' ') {
            mp[str[i]]++;
        }
    }
    
    //output
    for(auto i:mp) {
        cout << i.first << " = " << i.second << " ";
    }
    return 0;
}
```
